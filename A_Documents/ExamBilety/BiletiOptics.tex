\newcounter{rand}
\def\RandData#1{
\readrecordarray{#1}\data
\setcounter{rand}{\pdfuniformdeviate\dataROWS{}}
\ifnum\therand=0\stepcounter{rand}\fi
\data[\arabic{rand}]
}

\newcounter{datarow}
\def\GetData#1{
\readrecordarray{#1}\data
\stepcounter{datarow}
\data[\arabic{datarow}]
}


\newcommand\Repeat[2]{%
\def\filename{#1}
\readrecordarray{#1}\data
 \newcounter{i}
  \loop \ifnum\value{i} < \numexpr\nrows/3\relax
    #2%
    \stepcounter{i}%
  \repeat
}
